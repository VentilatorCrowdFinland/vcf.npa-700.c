148 static npa_ret_t parse_value (const npa_variant_t model, const uint8_t * const raw_data,
149  float * const pressure_pa)
150 {
151  npa_ret_t ret_code = NPA_SUCCESS;
152  // Mask status bits out.
153  const uint16_t OUT_U16 = ( ( (uint32_t) raw_data[0U] << 8U)
154  + (uint32_t) raw_data[1U])
155  & 0x3FFFU;
156  // Use floats in calculation
157  float Pmin = 0.0F;
158  float Pmax = 0.0F;
159  const float OUTmax = (float) NPA_PRES_MAX_NONSAT;
160  const float OUTmin = (float) NPA_PRES_MIN_NONSAT;
161  const float OUT = (float) OUT_U16;
162 
163  if (check_saturation (OUT_U16))
164  {
165  ret_code |= NPA_WARN_SAT;
166  }
167 
168  ret_code |= get_scaling (model, &Pmax, &Pmin);
169  /*
170  * Pressure can be calculated from the sensor output using the following formula:
171  *
172  * P = Pmin + (OUT - OUTmin) / (OUTmax - OUTmin) * (Pmax - Pmin)
173  */
174  *pressure_pa = Pmin + (OUT - OUTmin) / (OUTmax - OUTmin) * (Pmax - Pmin);
175  return ret_code;
176 }
